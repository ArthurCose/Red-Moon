print("\nlocals:")
local b = 12
do
  local a
  print(a)
  a = 9
  local b, c = 8, 7
  print(a, b, c)
  do
    local a, b, c = 3, 4
    print(a, b, c)
  end
  print(a, b, c)
end
print(a, b, c)
print(_ENV.a, _ENV.b, _ENV.c)

local foo = function(...) return ... end
local bar = function(a, b, c) return c end
